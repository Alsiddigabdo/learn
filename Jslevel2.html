شرح خصائص لغة JavaScript

في هذا المقال، سنتعرف على بعض الخصائص الأساسية في لغة JavaScript ونقدم أمثلة على كيفية استخدامها:

1. العناصر الأساسية:

JavaScript يحتوي على عدة عناصر أساسية تُسهّل كتابة البرامج، ومنها:

- الوظيفة الرئيسية main:

function main() {
    // اكتب الكود هنا
}

- التعليقات // و /* ... */:

// هنا تعليق في سطر واحد

/*
هنا تعليق
عدة
أسطر
*/

2. المتغيرات:

يمكن استخدام المتغيرات لتخزين القيم والبيانات المختلفة، ويتم إعلان المتغيرات باستخدام "var":

var name = "John";
var age = 30;

3. العمليات الحسابية:

يمكن القيام بالعمليات الحسابية الأساسية مثل الجمع والطرح والضرب والقسمة:

var sum = 2 + 2;
var difference = 5 - 3;
var multiplication = 4 * 4;
var division = 10 / 2;

4. هياكل التحكم:

يسمح لنا JavaScript بالتحكم في تنفيذ الأكواد باستخدام هياكل التحكم:

- الشرط if و else:

if (condition) {
    // الكود إذا تم اكتشاف صحة الشرط
} else {
    // الكود إذا كان الشرط غير صحيح
}

- الحلقة for و while:

for (var i = 0; i < 5; i++) {
    // اكتب الكود هنا الذي سيتم تنفيذه في كل دورة
}

var j = 0;
while (j < 5) {
    // اكتب الكود هنا الذي سيتم تنفيذه في كل دورة
    j++;
}

greet("John");

6. جمل التحكم:

تساعدنا جمل التحكم على تحديد تدفق التنفيذ في البرامج:

- الإيقاف المبكر break:

for (var i = 0; i < 10; i++) {
    if (i === 5) {
        break; // انهاء الحلقة إذا كانت قيمة i تساوي 5
    }
    console.log(i);
}

- السماح بالتنفيذ القادم continue:

for (var i = 0; i < 10; i++) {
    if (i === 5) {
        continue; // الانتقال إلى التكرار القادم إذا كانت قيمة i تساوي 5
    }
    console.log(i);
}

11. الاستجابة للعمليات الغير متزامنة:

يمكننا استخدام الوعود Promise ودوال async وawait للتعامل مع العمليات الغير متزامنة مثل جلب البيانات من خوادم الويب:

function fetchData() {
    return new Promise(function(resolve, reject) {
        // اكتب الكود هنا لجلب البيانات من خوادم الويب
        if (data) {
            resolve(data); // تنفيذ الوعد بجلب البيانات بنجاح
        } else {
            reject("Failed to fetch data"); // رفض الوعد في حالة فشل جلب البيانات
        }
    });
}

async function getData() {
    try {
        var result = await fetchData(); // استدعاء دالة غير متزامنة باستخدام كلمة الأمر await
        console.log(result);
    } catch (error) {
        console.log(error);
    }
}

getData();
